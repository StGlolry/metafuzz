          def input_to_bitstring( value )
                unless value.respond_to? :to_str  and value.to_str=~/^[01\s]*$/
                        raise ArgumentError, "BitstringField(#{@name}) (PARSE): Input value not a bitstring."
                end
                parse_buffer(value.to_str.gsub(/\s/,''))
          end

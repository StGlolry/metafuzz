    def parse_buffer_lazy( bitstring )
                    
            bitstring||=""
            unless bitstring.is_a? String and bitstring=~/^[10]*$/
                raise ArgumentError, "#{self.class.to_s[/\w+$/]} (#{@name}): <Internal> bitstring buffer borked??"
            end
            bitstring
    end

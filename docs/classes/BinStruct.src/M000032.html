        def to_s
                # splice in the separators

                separated_fields=self.fields.inject( [] ) {|arr, elem| arr.push self.separator unless arr.empty?; arr.push elem}
                bits=separated_fields.inject("") {|str,field| field.kind_of?(Fields::Field) ? str << field.bitstring : str << field.to_s.unpack('B*').to_s}
                unless bits.length % 8 == 0
                        puts "Warning, structure not byte aligned, right padding with 0"
                end
                bytearray=bits.scan(/.{1,8}/)
                bytearray << '0'*(8-bytearray.last.length) if bytearray.last.length < 8
                bytearray.map {|byte| "" << byte.to_i(2)}.join + @children.to_s
        end

                def initialize(series, start, step, limit, *transforms)
                        
                        @series=series
                        @transforms=transforms
                        @start,@step,@limit=start,step,limit

                        unless series.respond_to?(:to_a)
                                raise ArgumentError, "Err: object to repeat must respond to to_a"
                        end
                        if series.respond_to?(:to_a)
                                items=series.to_a
                        else
                                items=[series]
                        end
                        # Now that I think of it, I think it would be better to
                        # keep series as a generator, and convert other types over
                        # because that way we don't need to calculate every item all at once
                        # Fix Later.
                        results=[]
                        items.each {|item| 
                                Range.new(start,limit).step(step) {|repeats|
                                        tmp=[]
                                        repeats.times do
                                                tmp.push(item)
                                        end
                                        results.push(tmp) unless repeats ==0
                                }
                        }
                        @block=proc {|g|
                                results.each {|r|
                                        # run procs on array to be yielded
                                        g.yield(transforms.inject(r) {|r, xform|
                                                xform.call(r)
                                        })
                                }
                        }
                        @index = 0 
                        @queue = [] 
                        @cont_next = @cont_yield = @cont_endp = nil 
                        if @cont_next = callcc { |c| c } 
                                @block.call(self) 
                                @cont_endp.call(nil) if @cont_endp 
                        end 
                        self 
                end
